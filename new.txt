    /* ======================
       SVG ЗВЕЗДА
    ====================== */

    function createStar(filled = true) {

        return `
    <svg viewBox="0 0 576 512">
        <path class="${filled ? 'sm-star-full' : 'sm-star-empty'}"
        d="M381.2 150.3L524.9 171.5C569.3 178.1 586.7 231.1 554.7 262.8L449.8 366.3L475.2 509.5C482.9 553.8 436.5 587.5 397.8 566.6L288 510.8L178.2 566.6C139.5 587.5 93.1 553.8 100.8 509.5L126.2 366.3L21.3 262.8C-10.7 231.1 6.7 178.1 51.1 171.5L194.8 150.3L259.9 17.8C279.4 -22.6 336.6 -22.6 356.1 17.8L381.2 150.3Z"/>
    </svg>`;
    }


    /* ======================
       ГЕНЕРАЦИЯ ЗВЕЗД
    ====================== */

    function renderStars(rating) {

        let stars = '';

        for (let i = 1; i <= 5; i++) {

            if (i <= rating) {
                stars += createStar(true);
            } else {
                stars += createStar(false);
            }
        }

        return stars;
    }


    /* ======================
       РЕНДЕР ОТЗЫВОВ
    ====================== */

    function renderReviews() {

        const container = document.getElementById('sm-reviews');

        reviewsData.forEach((review, index) => {

            const reviewEl = document.createElement('div');
            reviewEl.className = 'sm-review';

            reviewEl.innerHTML = `

                <div class="sm-review-avatar" style="background-image:url('${review.avatar}')">
                </div>

            <div class="sm-review-content">

                <div class="sm-review-top">

                    <div>
                        <div class="sm-review-name">${review.name}</div>
                        <div class="sm-review-date">${review.date}</div>
                    </div>

                    <div class="sm-review-stars">
                        ${renderStars(review.rating)}
                    </div>

                </div>

                <div class="sm-review-text" id="sm-text-${index}">
                    ${review.text}
                </div>

                <button class="sm-read-more" id="sm-btn-${index}">
                        Показать полностью
                </button>

            </div>
        `;

            container.appendChild(reviewEl);

            setupReadMore(index);
        });
    }


    /* ======================
       КНОПКА "ПОКАЗАТЬ"
    ====================== */

    function setupReadMore(index) {

        const text = document.getElementById(`sm-text-${index}`);
        const btn = document.getElementById(`sm-btn-${index}`);

        // Проверяем — обрезан ли текст
        if (text.scrollHeight <= 70) {
            btn.style.display = 'none';
            return;
        }

        btn.addEventListener('click', () => {

            text.classList.toggle('sm-expanded');

            btn.textContent =
                text.classList.contains('sm-expanded')
                    ? 'Скрыть'
                    : 'Показать полностью';
        });
    }


    /* ======================
       ЗАПУСК
    ====================== */

    renderReviews();
